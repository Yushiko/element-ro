//= NPC HECHO A PETCION DE ~noteinteresa~
//=========================================================

morocc.gat,179,98,7	script	Pet~Friendly	121,{

// Variables
set	@name$,"[Pet Friendly]";	//Nombre del NPC
set	@precio1,1000000;		// Costo de Friednly
set	@precio2,20000;			// Costo de Food
set 	$Cobrar,2;			// Cobro -Si- (1) -No- (2)

// Script
	mes @name$;
	mes "Hola, yo puedo aumentar la confianza de tu Pet,";
	mes "Ademas de darle de comer";
	if ($Cobrar == 1) mes "Todo por un ^5555FF" + @precio1 + "^000000 Zenys";
	next;
	mes "¿Que deseas hacer?";
	menu "Aumentar la Confianza",petup,"Darle de Comer",petfood,"No, gracias",no;

petup:
	if ($Cobrar == 1) if(zeny <= @precio1) goto no_zeny;
	next;
	mes @name$;
	mes "Bien, veamos...";
	if ($Cobrar == 1) set zeny, zeny-@precio1;
	next;
	atcommand strcharinfo(0)+":@petfriendly " +100;
	next;	
	mes "Listo, ya te quiere mucho.";
	close;

petfood:
	next;
	if ($Cobrar == 1) if(zeny <= @precio2) goto no_zeny;
	mes @name$;
	mes "Bien, veamos...";
	if ($Cobrar == 1) set zeny, zeny-@precio2;
	next;
	atcommand strcharinfo(0)+":@pethungry " +100;
	next;	
	mes "Listo, ya esta satisfecho.";
	emotion e_no1;
	close;

no:	
	mes @name$;
	mes "Ok. nos vemos,";
	mes "pero pronto me necesitaras";
	emotion e_gg;
	close;

no_zeny:
	mes @name$;
	mes "¬¬U no tienes dinero, regresa cuanto lo tengas.";
	emotion e_omg;
	close;
}
